 ~~~CSS
 input,button {
  display:block;
  margin: 1em;
}  
~~~

~~~javascript
 function obterMedia() {
  var a = parseInt(document.getElementById("a").value);
  var b = parseInt(document.getElementById("b").value);
  var c = parseInt(document.getElementById("c").value);

  var media = (a + b + c)/3;
  document.getElementById("resultado").innerHTML = "resultado: " + media;
  }
~~~
